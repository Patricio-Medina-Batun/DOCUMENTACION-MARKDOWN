\begin{pmatrix}

\begin{pmatrix}

\end{pmatrix}
$$

\begin{pmatrix}

\end{pmatrix}
$$

\end{pmatrix}

$$ A * B =
\begin{pmatrix}
4 & 6 \\
10 & 12 \\
\end{pmatrix}
$$
